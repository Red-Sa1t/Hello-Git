		```c
		void Save() {
		    FILE* fp = fopen("save.bin", "wb");
		    if (fp == NULL)
		        return;
		    fwrite(&num, sizeof(int), 1, fp);
		    fwrite(student,sizeof(Student), num, fp);
		    fclose(fp);
		    printf("保存成功!\n");
		}
		
		void Load() {
		    FILE* f = fopen("save.bin", "rb");
		    if (f == NULL)
		        return;
		    fread(&num, sizeof(int), 1, f);
		    fread(student, sizeof(Student), num, f);
		    fclose(f);
		    printf("载入成功!\n");
		}
		```
